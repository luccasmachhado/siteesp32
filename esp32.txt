#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <PubSubClient.h>

// =========================================
// WIFI
// =========================================
const char* WIFI_SSID = "MOB-ROBOTICA SUSTENTAVEL";
const char* WIFI_PASSWORD = "robotica3007";

// =========================================
// MQTT HIVEMQ CLOUD
// =========================================
const char* MQTT_BROKER =
"fff567fb3b6e42cabc200996cee91ce1.s1.eu.hivemq.cloud";

const int MQTT_PORT = 8883;

const char* MQTT_USERNAME = "reflector";
const char* MQTT_PASSWORD = "Reflector1";

const char* MQTT_CLIENT_ID = "esp32-reflector";

// =========================================
// TOPICO
// =========================================
const char* TOPIC = "reflector";

// =========================================
// PINO
// =========================================
const int LED_PIN = 23;

// =========================================
// OBJETOS
// =========================================
WiFiClientSecure espClient;
PubSubClient mqttClient(espClient);

// =========================================
// WIFI
// =========================================
void connectWiFi() {

  Serial.print("Conectando WiFi");

  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

  while (WiFi.status() != WL_CONNECTED) {

    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWiFi conectado!");

  Serial.print("IP: ");
  Serial.println(WiFi.localIP());
}

// =========================================
// CALLBACK MQTT
// =========================================
void mqttCallback(char* topic, byte* payload, unsigned int length) {

  Serial.println("Mensagem recebida!");

  String message = "";

  for (unsigned int i = 0; i < length; i++) {
    message += (char)payload[i];
  }

  message.trim();

  Serial.print("Topico: ");
  Serial.println(topic);

  Serial.print("Mensagem: ");
  Serial.println(message);

  if (message == "ON") {

    digitalWrite(LED_PIN, HIGH);

    Serial.println("LED LIGADO");
  }

  else if (message == "OFF") {

    digitalWrite(LED_PIN, LOW);

    Serial.println("LED DESLIGADO");
  }

  else {

    Serial.println("Comando invalido");
  }
}

// =========================================
// MQTT
// =========================================
void connectMQTT() {

  while (!mqttClient.connected()) {

    Serial.print("Conectando MQTT... ");

    bool connected = mqttClient.connect(
      MQTT_CLIENT_ID,
      MQTT_USERNAME,
      MQTT_PASSWORD
    );

    if (connected) {

      Serial.println("CONECTADO!");

      bool subscribed = mqttClient.subscribe(TOPIC);

      if (subscribed) {
        Serial.println("Subscribe OK");
      } else {
        Serial.println("Erro subscribe");
      }

    } else {

      Serial.print("Falha MQTT. Codigo: ");
      Serial.println(mqttClient.state());

      delay(5000);
    }
  }
}

// =========================================
// SETUP
// =========================================
void setup() {

  Serial.begin(115200);

  pinMode(LED_PIN, OUTPUT);

  digitalWrite(LED_PIN, LOW);

  connectWiFi();

  // Ignora certificado SSL
  espClient.setInsecure();

  mqttClient.setServer(MQTT_BROKER, MQTT_PORT);

  mqttClient.setCallback(mqttCallback);

  connectMQTT();
}

// =========================================
// LOOP
// =========================================
void loop() {

  if (!mqttClient.connected()) {
    connectMQTT();
  }

  mqttClient.loop();
}
